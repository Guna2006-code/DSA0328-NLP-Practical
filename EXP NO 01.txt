import re
text = "My phone number is 9876543210 and my email is test123@gmail.com"
match_result = re.match(r"My", text)
if match_result:
    print("Match found at the beginning!")
else:
    print("No match at the beginning.")
search_result = re.search(r"\d+", text)
if search_result:
    print("First number found:", search_result.group())
numbers = re.findall(r"\d+", text)
print("All numbers:", numbers)
email = re.search(r"\S+@\S+", text)
if email:
    print("Email found:", email.group())
new_text = re.sub(r"\d", "X", text)
print("Text after replacing digits:", new_text)
