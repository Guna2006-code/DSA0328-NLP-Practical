def plural_fsm(word):
    state = "q0" 

    if word.endswith(("s", "x", "z", "ch", "sh")):
        state = "q1"
    elif word.endswith("y") and word[-2] not in "aeiou":
        state = "q2"
    else:
        state = "q3"

    if state == "q1":
        return word + "es"
    elif state == "q2":
        return word[:-1] + "ies"
    elif state == "q3":
        return word + "s"

words = ["cat", "bus", "box", "city", "boy", "brush"]

for w in words:
    print(w, "→", plural_fsm(w))
